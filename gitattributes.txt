# Normalize line endings across platforms.
# Text files are stored as LF in the repository; reviewers on any OS get LF.
* text=auto eol=lf

# Explicit text types (code, config, docs)
*.py    text eol=lf
*.md    text eol=lf
*.yaml  text eol=lf
*.yml   text eol=lf
*.txt   text eol=lf
*.sh    text eol=lf
*.json  text eol=lf
*.csv   text eol=lf

# Binary types — never normalize
*.png   binary
*.jpg   binary
*.pdf   binary
*.xpt   binary
*.joblib binary
*.pkl   binary
*.parquet binary
